void ToggleRenderersAndColliders(Transform parent, bool enable)
    {
        // Attiva o disattiva il renderer e il collider dell'oggetto corrente, se presenti
        Renderer renderer = parent.GetComponent<Renderer>();
        Collider collider = parent.GetComponent<Collider>();

        if (renderer != null)
            renderer.enabled = enable;

        if (collider != null)
            collider.enabled = enable;

        // Richiama la funzione ricorsivamente per tutti i figli
        for (int i = 0; i < parent.childCount; i++)
        {
            Transform child = parent.GetChild(i);
            ToggleRenderersAndColliders(child, enable);
        }
    }
}
